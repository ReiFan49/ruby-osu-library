end

def inherited(cls)
  cls.class_exec do
    def entry=(cls)
      fail TypeError, "expected cls is a class!" unless cls.is_a?(Class)
      fail TypeError, "expected cls to Entry class!" unless cls <= Entry
      @_entry = cls
    end
  end
end

def to_s
  sprintf("[%1$s]%2$s%3$s",
    @name, $/,
    @contents.join($/)
  )
end
